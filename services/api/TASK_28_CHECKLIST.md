## üîç Verification Queries

```sql
-- Check all data
SELECT 'Categories' as type, COUNT(*) as count FROM categories
UNION ALL
SELECT 'NFT Catalog', COUNT(*) FROM nft_catalog
UNION ALL
SELECT 'Questions', COUNT(*) FROM questions
UNION ALL
SELECT 'Active Seasons', COUNT(*) FROM seasons WHERE is_active = true;

-- Check NFTs per category
SELECT c.name, COUNT(n.id) as nft_count
FROM categories c
LEFT JOIN nft_catalog n ON n.category_id = c.id
GROUP BY c.name
ORDER BY c.name;

-- Check questions per category
SELECT c.name, COUNT(q.id) as question_count
FROM categories c
LEFT JOIN questions q ON q.category_id = c.id
GROUP BY c.name
ORDER BY c.name;

-- Check season details
SELECT id, name, starts_at, ends_at, grace_days, is_active
FROM seasons
WHERE is_active = true;
```
